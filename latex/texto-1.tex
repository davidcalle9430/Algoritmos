% Algoritmo 1


\begin{algorithm}[htp]
\DontPrintSemicolon
\SetAlgoLined
\SetKwInOut{Input}{Entrada}\SetKwInOut{Output}{Salida}
\Input{Una secuencia S de numeros, Entero v}
\Output{$i\mid$  i $\in$  $\mathbb{N}$ $\land$ ( i $>$ 0   $\lor$  i $<$ $|$ S $|$  $\lor$ i = $\aleph$ ) }
\BlankLine

\For{ $i\leftarrow 1$ to  $|$ S $|$  }{
	\If{S[i] = v}{
		return i\;
	}
	return $\aleph$\;
}

\caption{Busqueda Lineal}
\end{algorithm}
